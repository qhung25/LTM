using System;
using System.Drawing;
using System.Net;
using System.Net.Sockets;
using System.Text;
using System.Threading;
using System.Windows.Forms;

// Thêm partial để tránh lỗi Designer nếu bạn dùng Visual Studio
public partial class NewTcpChat : Form
{
    private static TextBox newText;
    private static ListBox results;
    private static ListBox hosts;
    private static Socket client;
    private static byte[] data = new byte[1024];

    public NewTcpChat()
    {
        Text = "New TCP Chat Program";
        Size = new Size(400, 420);

        Label label1 = new Label();
        label1.Parent = this;
        label1.Text = "Enter text string:";
        label1.AutoSize = true;
        label1.Location = new Point(10, 30);

        newText = new TextBox();
        newText.Parent = this;
        newText.Size = new Size(200, 20);
        newText.Location = new Point(10, 55);

        results = new ListBox();
        results.Parent = this;
        results.Location = new Point(10, 85);
        results.Size = new Size(360, 150);

        Label label2 = new Label();
        label2.Parent = this;
        label2.Text = "Active hosts (Select one to connect):";
        label2.AutoSize = true;
        label2.Location = new Point(10, 245);

        hosts = new ListBox();
        hosts.Parent = this;
        hosts.Location = new Point(10, 265);
        hosts.Size = new Size(360, 80);

        Button sendit = new Button();
        sendit.Parent = this;
        sendit.Text = "Send";
        sendit.Location = new Point(220, 52);
        sendit.Click += new EventHandler(ButtonSendOnClick);

        Button connect = new Button();
        connect.Parent = this;
        connect.Text = "Connect";
        connect.Location = new Point(295, 20);
        connect.Click += new EventHandler(ButtonConnectOnClick);

        Button listen = new Button();
        listen.Parent = this;
        listen.Text = "Listen";
        listen.Location = new Point(295, 52);
        listen.Click += new EventHandler(ButtonListenOnClick);

        // Luồng tìm kiếm các máy đang online
        Thread fh = new Thread(new ThreadStart(findHosts));
        fh.IsBackground = true;
        fh.Start();
    }

    // Hàm bổ trợ để ghi log vào ListBox từ bất kỳ luồng nào
    private void UpdateResults(string message)
    {
        if (results.InvokeRequired)
        {
            results.Invoke(new Action(() => results.Items.Add(message)));
        }
        else
        {
            results.Items.Add(message);
        }
    }

    void ButtonListenOnClick(object obj, EventArgs ea)
    {
        UpdateResults("Listening for a client...");
        Socket newsock = new Socket(AddressFamily.InterNetwork, SocketType.Stream, ProtocolType.Tcp);
        IPEndPoint iep = new IPEndPoint(IPAddress.Any, 9050);
        try
        {
            newsock.Bind(iep);
            newsock.Listen(5);
newsock.BeginAccept(new AsyncCallback(AcceptConn), newsock);

            Thread advertise = new Thread(new ThreadStart(srvrAdvertise));
            advertise.IsBackground = true;
            advertise.Start();
        }
        catch (Exception ex) { MessageBox.Show(ex.Message); }
    }

    void ButtonConnectOnClick(object obj, EventArgs ea)
    {
        if (hosts.SelectedItem == null)
        {
            MessageBox.Show("Vui lòng chọn một host trong danh sách Active hosts!");
            return;
        }

        UpdateResults("Connecting...");
        client = new Socket(AddressFamily.InterNetwork, SocketType.Stream, ProtocolType.Tcp);

        string selectedhost = (string)hosts.SelectedItem;
        // Tách chuỗi để lấy IP (Ví dụ: "MyPC:192.168.1.5:9051" -> lấy phần ở giữa)
        string[] hostarray = selectedhost.Split(':');
        IPEndPoint iep = new IPEndPoint(IPAddress.Parse(hostarray[1]), 9050);

        client.BeginConnect(iep, new AsyncCallback(Connected), client);
    }

    void ButtonSendOnClick(object obj, EventArgs ea)
    {
        // Chống lỗi NullReferenceException nếu chưa kết nối
        if (client == null || !client.Connected)
        {
            MessageBox.Show("Chưa có kết nối nào được thiết lập!");
            return;
        }

        byte[] message = Encoding.ASCII.GetBytes(newText.Text);
        newText.Clear();
        client.BeginSend(message, 0, message.Length, 0, new AsyncCallback(SendData), client);
    }

    void AcceptConn(IAsyncResult iar)
    {
        Socket oldserver = (Socket)iar.AsyncState;
        client = oldserver.EndAccept(iar);
        UpdateResults("Connection from: " + client.RemoteEndPoint.ToString());

        Thread receiver = new Thread(new ThreadStart(ReceiveData));
        receiver.IsBackground = true;
        receiver.Start();
    }

    void Connected(IAsyncResult iar)
    {
        try
        {
            client.EndConnect(iar);
            UpdateResults("Connected to: " + client.RemoteEndPoint.ToString());

            Thread receiver = new Thread(new ThreadStart(ReceiveData));
            receiver.IsBackground = true;
            receiver.Start();
        }
        catch (Exception)
        {
            UpdateResults("Error connecting");
        }
    }

    void SendData(IAsyncResult iar)
    {
        Socket remote = (Socket)iar.AsyncState;
        int sent = remote.EndSend(iar);
    }

    void ReceiveData()
    {
        int recv;
        string stringData;
        try
        {
            while (true)
            {
                recv = client.Receive(data);
                if (recv == 0) break; // Kết nối bị đóng

                stringData = Encoding.ASCII.GetString(data, 0, recv);
                if (stringData == "bye") break;

                UpdateResults(stringData);
            }
        }
        catch { }

        UpdateResults("Connection stopped");
        client.Close();
    }
void srvrAdvertise()
    {
        Socket server = new Socket(AddressFamily.InterNetwork, SocketType.Dgram, ProtocolType.Udp);
        server.SetSocketOption(SocketOptionLevel.Socket, SocketOptionName.Broadcast, 1);
        IPEndPoint iep = new IPEndPoint(IPAddress.Broadcast, 9051);
        byte[] hostname = Encoding.ASCII.GetBytes(Dns.GetHostName());

        while (true)
        {
            server.SendTo(hostname, iep);
            Thread.Sleep(5000); // Gửi quảng bá mỗi 5 giây thay vì 60s để nhanh thấy máy hơn
        }
    }

    void findHosts()
    {
        // Bind một lần duy nhất bên ngoài vòng lặp
        Socket remoteHosts = new Socket(AddressFamily.InterNetwork, SocketType.Dgram, ProtocolType.Udp);
        IPEndPoint iep = new IPEndPoint(IPAddress.Any, 9051);
        remoteHosts.Bind(iep);

        while (true)
        {
            EndPoint ep = (EndPoint)iep;
            byte[] dataRecv = new byte[1024];
            int recv = remoteHosts.ReceiveFrom(dataRecv, ref ep);

            string stringData = Encoding.ASCII.GetString(dataRecv, 0, recv);
            string entry = stringData + ":" + ep.ToString();

            // Dùng Invoke để cập nhật ListBox hosts
            hosts.Invoke(new Action(() => {
                if (!hosts.Items.Contains(entry))
                    hosts.Items.Add(entry);
            }));
        }
    }

    [STAThread]
    public static void Main()
    {
        Application.EnableVisualStyles();
        Application.Run(new NewTcpChat());
    }
}
